To use this layout, first include the theme's CSS:

CSS

<link rel="stylesheet" href="css/style.css">

Then include the JavaScript file and initialize the Sidebar. Note that the JavaScript does not initialize automatically—you must explicitly create a new Sidebar instance:

JavaScript

<script type="application/javascript" src="/js/asciiart.min.js"></script>
<script type="application/javascript">
    document.addEventListener("DOMContentLoaded", () => {
        // Initialize the Sidebar component
        new AsciiartTheme.Sidebar("#sidebar");
    });
</script>

Alternatively, you can use an ES6 module to import and initialize the Sidebar component:

ES6 Module

<script type="component">
    import { Sidebar } from "js/asciiart.esm.min.js";

    document.addEventListener("DOMContentLoaded", () => {
        // Initialize the Sidebar using ES6 module
        new Sidebar("#sidebar");
    });
</script>
